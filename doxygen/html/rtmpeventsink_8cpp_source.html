00041         //1. Save the stream name
00042         if (_streamName == streamName)
00043                 return true;

00048         //2. Get the RTMP protocol

00050         if (pProtocol == NULL) {
00051                 FATAL("Unable to get the RTMP protocol");
00052                 return false;

00054 
00055         //3. Prepare the invoke
00056         Variant parameters;
00057         parameters.PushToArray(Variant());
00058         parameters.PushToArray(streamName);
00059         Variant request = GenericMessageFactory::GetInvoke(3, 0, 0, false, 0,
00060                         "streamAvailable", parameters);
00061 
00062         //4. Send it
00063         if (!pProtocol->SendMessage(request)) {
00064                 FATAL("Unable to send RTMP message");
00065                 pProtocol->EnqueueForDelete();

00067         }
00068 
00069         //5. Done
00070         return true;
