00059                 WARN("Unable to read 1 byte");
00060                 return false;
00061         }

00063                 case 3:
00064                 {
00065                         _unsynchronisation = ((flags >> 7) != 0);
00066                         _extendedHeader = (((flags >> 6)&0x01) != 0);
00067                         _experimentalIndicator = (((flags >> 5)&0x01) != 0);
00068                         break;
00069                 }
00070                 case 4:
00071                 {
00072                         _unsynchronisation = ((flags >> 7) != 0);
00073                         _extendedHeader = (((flags >> 6)&0x01) != 0);
00074                         _experimentalIndicator = (((flags >> 5)&0x01) != 0);
00075                         _footerPresent = (((flags >> 4)&0x01) != 0);
00076                         break;
00077                 }
00078                 default:
00079                 {
00080                         WARN("Invalid version: ID3v%u.%u", _majorVersion, _minorVersion);
00081                         return false;
00082                 }
00083         }
00084 
00085 
00086         //2. Read the size
00087         uint8_t tempBuffer[4];
00088         if (!file.ReadBuffer(tempBuffer, 4)) {
00089                 WARN("Unable to read 4 bytes");
00090                 return false;
00091         }
00092         uint32_t totalSize = (tempBuffer[0]&0x7f);
00093         totalSize = (totalSize << 7) | (tempBuffer[1]&0x7f);
00094         totalSize = (totalSize << 7) | (tempBuffer[2]&0x7f);
00095         totalSize = (totalSize << 7) | (tempBuffer[3]&0x7f);
00096 
00097         //3. See if we go to the end of the ID3 we have a valid media frame
00098         //After that we go back to where we left (end of the header, offset 10)
00099         if (!file.SeekTo(totalSize + 10)) {
00100                 WARN("Unable to seek to 0x%x offset", totalSize);
00101                 return false;
00102         }
00103 
00104         if (!file.ReadBuffer(tempBuffer, 2)) {
00105                 WARN("Unable to read 2 bytes");
00106                 return false;
00107         }
00108         if ((tempBuffer[0] != 0xff) || ((tempBuffer[1] >> 5) != 7)) {
00109                 WARN("Unable to position on frame data.");
00110                 return false;
00111         }
00112         if (!file.SeekTo(10)) {
00113                 WARN("Unable to seek to 0x0a offset");
00114                 return false;
00115         }
00116 
00117         //4. Skip the extended header
00118         if (_extendedHeader) {
00119                 NYIR;
00120         }
00121 
00122         //5. Store the tags inside a IOBuffer
00123         IOBuffer buffer;
00124         if (!buffer.ReadFromFs(file, totalSize)) {
00125                 WARN("Unable to read %u bytes", totalSize);
00126                 return false;
00127         }
00128 
00129         if (!ParseTags(buffer)) {
00130                 WARN("Unable to parse the tags. Metadata might be incomplete");
00131         }
