00299                 PIDDescriptor &pidDescriptor, uint8_t *pBuffer, uint32_t &cursor,
00300                 uint32_t maxCursor) {
00301         //1. Advance the pointer field
00302         if (TS_TRANSPORT_PACKET_IS_PAYLOAD_START(packetHeader)) {
00303                 CHECK_BOUNDS(1);
00304                 CHECK_BOUNDS(pBuffer[cursor]);
00305                 cursor += pBuffer[cursor] + 1;
00306         }
00307 
00308         //1. Get the crc from the packet and compare it with the last crc.
00309         //if it is the same, ignore this packet
00310         uint32_t crc = TSPacketPAT::PeekCRC(pBuffer, cursor, maxCursor);
00311         if (crc == 0) {
00312                 FATAL("Unable to read crc");
00313                 return false;
00314         }
00315         if (pidDescriptor.payload.crc == crc) {
00316                 return true;
00317         }
00318 
00319         //2. read the packet
00320         TSPacketPAT packetPAT;
00321         if (!packetPAT.Read(pBuffer, cursor, maxCursor)) {
00322                 FATAL("Unable to read PAT");
00323                 return false;
00324         }
00325 
00326         //3. Store the crc
00327         pidDescriptor.payload.crc = packetPAT.GetCRC();

00330         //4. Store the pid types found in PAT
00331 
00332         FOR_MAP(packetPAT.GetPMTs(), uint16_t, uint16_t, i) {

00334                 pDescr->pid = MAP_VAL(i);

00341                 PIDDescriptor *pDescr = new PIDDescriptor;

00343                 pDescr->type = PID_TYPE_NIT;
