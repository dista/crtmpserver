00031 bool InboundSSLProtocol::InitGlobalContext(Variant &parameters) {
00032         //1. Comput the hash on key/cert pair and see
00033         //if we have a global context with that hash
00034         string hash = "";
00035         if (parameters["hash"] != V_STRING) {
00036                 if (parameters[CONF_SSL_KEY] != V_STRING
00037                                 || parameters[CONF_SSL_CERT] != V_STRING) {
00038                         FATAL("No key/cert provided");
00039                         return false;
00040                 }
00041                 hash = md5((string) parameters[CONF_SSL_KEY]
00042                                 + (string) parameters[CONF_SSL_CERT], true);
00043                 parameters["hash"] = hash;
00044         } else {
00045                 hash = (string) parameters["hash"];
00046         }
00047         string key = parameters[CONF_SSL_KEY];
00048         string cert = parameters[CONF_SSL_CERT];
00049         _pGlobalSSLContext = _pGlobalContexts[hash];
00050 
00051         //2. Initialize the global context based on the specified
00052         //key/cert pair if we don't have it
00053         if (_pGlobalSSLContext == NULL) {
00054                 //3. prepare the global ssl context
00055                 _pGlobalSSLContext = SSL_CTX_new(SSLv23_method());
00056                 if (_pGlobalSSLContext == NULL) {
00057                         FATAL("Unable to create global SSL context");
00058                         return false;
00059                 }
00060 
00061                 //4. setup the certificate
00062                 if (SSL_CTX_use_certificate_file(_pGlobalSSLContext, STR(cert),
00063                                 SSL_FILETYPE_PEM) <= 0) {
00064                         FATAL("Unable to load certificate %s; Error(s) was: %s",
00065                                         STR(cert),
00066                                         STR(GetSSLErrors()));
00067                         SSL_CTX_free(_pGlobalSSLContext);
00068                         _pGlobalSSLContext = NULL;
00069                         return false;
00070                 }
00071 
00072                 //5. setup the private key
00073                 if (SSL_CTX_use_PrivateKey_file(_pGlobalSSLContext, STR(key),
00074                                 SSL_FILETYPE_PEM) <= 0) {
00075                         FATAL("Unable to load key %s; Error(s) was: %s",
00076                                         STR(key),
00077                                         STR(GetSSLErrors()));
00078                         SSL_CTX_free(_pGlobalSSLContext);
00079                         _pGlobalSSLContext = NULL;
00080                         return false;
00081                 }
00082 
00083                 //6. disable client certificate authentication
00084                 SSL_CTX_set_verify(_pGlobalSSLContext, SSL_VERIFY_NONE, NULL);
00085 
00086                 //7. Store the global context for later usage
00087                 _pGlobalContexts[hash] = _pGlobalSSLContext;
