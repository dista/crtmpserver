01580         result.Reset();

01583         vector<string> parts;
01584         split(raw, ";", parts);
01585 
01586         //2. Construct the result
01587         for (uint32_t i = 0; i < parts.size(); i++) {
01588                 string part = parts[i];
01589                 trim(part);
01590                 if (part == "")
01591                         continue;
01592                 string::size_type pos = part.find('=');
01593                 if (pos == string::npos) {
01594                         result[lowerCase(part)] = (bool)true;
01595                         continue;
01596                 }
01597                 result[lowerCase(part.substr(0, pos))] = part.substr(pos + 1);
01598         }
01599 
01600         vector<string> keys;
01601         ADD_VECTOR_END(keys, "client_port");
01602         ADD_VECTOR_END(keys, "server_port");
01603         ADD_VECTOR_END(keys, "interleaved");
01604 
01605         for (uint32_t i = 0; i < keys.size(); i++) {
01606                 string key = keys[i];
01607                 if (!result.HasKey(key))
01608                         continue;
01609                 parts.clear();
01610                 raw = (string) result[key];
01611                 split(raw, "-", parts);
01612                 if (parts.size() != 2) {
01613                         FATAL("Invalid transport line: %s", STR(raw));
01614                         return false;
01615                 }
01616                 uint16_t data = atoi(STR(parts[0]));
01617                 uint16_t rtcp = atoi(STR(parts[1]));
01618                 if (((data % 2) != 0) || ((data + 1) != rtcp)) {
01619                         FATAL("Invalid transport line: %s", STR(raw));
01620                         return false;
01621                 }
01622                 string all = format("%"PRIu16"-%"PRIu16, data, rtcp);
01623                 if (all != raw) {
01624                         FATAL("Invalid transport line: %s", STR(raw));
01625                         return false;
01626                 }

01628                 result[key]["data"] = (uint16_t) data;
