00052 bool BaseOutStream::Link(BaseInStream *pInStream, bool reverseLink) {
00053         if ((!pInStream->IsCompatibleWithType(GetType()))
00054                         || (!IsCompatibleWithType(pInStream->GetType()))) {
00055                 FATAL("stream type %s not compatible with stream type %s",
00056                                 STR(tagToString(GetType())),
00057                                 STR(tagToString(pInStream->GetType())));
00058                 return false;
00059         }
00060         if (_pInStream != NULL) {
00061                 if (_pInStream->GetUniqueId() == pInStream->GetUniqueId()) {
00062                         WARN("BaseOutStream::Link: This stream is already linked");
00063                         return true;
00064                 }
00065                 FATAL("BaseOutStream::Link: This stream is already linked to stream with unique id %u",
00066                                 _pInStream->GetUniqueId());
00067                 return false;
00068         }
00069         _pInStream = pInStream;
00070         if (reverseLink) {
00071                 if (!_pInStream->Link(this, false)) {
00072                         FATAL("BaseOutStream::Link: Unable to reverse link");
00073                         _pInStream = NULL;
00074                         return false;
00075                 }
00076         }
00077         SignalAttachedToInStream();
00078         return true;
00079 }
