00175                 return SendAuthRequired(auth);

00178         if (authorization.length() < 7) {
00179                 FATAL("Invalid authorization header");
00180                 return SendAuthRequired(auth);

00183         map<string, string> authMap = mapping(authorization, ",", "=", true);
00184 
00185         FOR_MAP(authMap, string, string, i) {
00186                 replace(MAP_VAL(i), "\"", "");

00188

00190         if (!auth["realms"].HasKey(authMap["realm"])) {
00191                 FATAL("Invalid authorization header");
00192                 return SendAuthRequired(auth);
00193         }
00194         if (!auth["realms"][authMap["realm"]]["credentials"].HasKey(authMap["username"])) {
00195                 FATAL("Invalid authorization header");

00202                         + ":" + authMap["nonce"] + ":" + authMap["nc"] + ":" + authMap["cnonce"] + ":" + authMap["qop"] + ":"
