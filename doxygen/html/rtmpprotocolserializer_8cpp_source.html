00209                 }
00210         }
00211

00217

00219         VH_ML(message) = GETAVAILABLEBYTESCOUNT(_internalBuffer);

00236                         buffer.ReadFromInputBuffer(&_internalBuffer, 0, chunkSize);
00237                         channel.lastOutProcBytes += chunkSize;
00238                         _internalBuffer.Ignore(chunkSize);

00241                         channel.lastOutProcBytes += available;
00242                         _internalBuffer.Ignore(available);

00244         }
00245         channel.lastOutProcBytes = 0;
