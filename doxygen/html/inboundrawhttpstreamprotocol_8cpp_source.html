00084 
00085         if (!_streamNameAcquired) {
00086                 return true;

00088

00090         map<uint32_t, BaseStream *> inStreams =
00091                         GetApplication()->GetStreamsManager()->FindByTypeByName(
00092                         ST_IN_NET_RAW, _streamName, false, true);

00094                 if (lowerCase(_streamName) == "crossdomain.xml") {
00095                         return SendCrossDomain();

00097                         FATAL("Stream %s not found", STR(_streamName));
00098                         return Send404NotFound();
00099                 }
00100         }
00101         BaseInStream *pInStream = (BaseInStream *) MAP_VAL(inStreams.begin());
00102 
00103         //8. Create our raw outbound stream
00104         _pOutStream = new OutNetRawStream(this,
00105                         GetApplication()->GetStreamsManager(), _streamName);

00150 bool InboundRawHTTPStreamProtocol::AcquireStreamName(IOBuffer &buffer) {
00151         //1. Get the buffer and length
00152         uint32_t length = GETAVAILABLEBYTESCOUNT(buffer);
00153         if (length < 16) {
00154                 return true;
00155         }
00156         uint8_t *pBuffer = GETIBPOINTER(buffer);
00157 
00158 
00159         //2. search for the first \r\n sequence
00160         bool found = false;
00161         for (uint32_t i = 0; i < length - 1; i++) {
00162                 //3. Limit the search to 1024 bytes
00163                 if (i >= 1024) {
00164                         FATAL("HTTP first line request too long");
00165                         return false;
00166                 }
00167                 if ((pBuffer[i] == '\r') && (pBuffer[i + 1] == '\n')) {
00168                         pBuffer[i] = 0;
00169                         found = true;
00170                         break;
00171                 }
00172         }
00173         if (!found) {
00174                 return true;
00175         }
00176 
00177         //3. Get the first line
00178         string rawLine = (char *) pBuffer;
00179 
00180         //4. Split it into components delimited by space characters
00181         vector<string> parts;
00182         split(rawLine, " ", parts);
00183 
00184         //5. validate the parts
00185         if (parts.size() != 3) {
00186                 FATAL("Invalid first line: %s", STR(rawLine));
00187                 return false;
00188         }
00189         if (parts[0] != HTTP_METHOD_GET) {
00190                 FATAL("Invalid first line: %s", STR(rawLine));
00191                 return false;
00192         }
00193         if (parts[2] != HTTP_VERSION_1_1) {
00194                 FATAL("Invalid first line: %s", STR(rawLine));
00195                 return false;
00196         }
00197         if (parts[1].length() < 2) {
00198                 FATAL("Invalid first line: %s", STR(rawLine));
00199                 return false;
00200         }
00201         if (parts[1][0] != '/') {
00202                 FATAL("Invalid first line: %s", STR(rawLine));
00203                 return false;
00204         }
00205 
00206         //6. Pick up the streamName
00207         _streamName = parts[1].substr(1);
