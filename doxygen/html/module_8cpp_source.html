00114 bool Module::BindAcceptor(Variant &node) {
00115         //1. Get the chain
00116         vector<uint64_t> chain;
00117         chain = ProtocolFactoryManager::ResolveProtocolChain(node[CONF_PROTOCOL]);
00118         if (chain.size() == 0) {
00119                 WARN("Invalid protocol chain: %s", STR(node[CONF_PROTOCOL]));
00120                 return true;
00121         }
00122 
00123         //2. Is it TCP or UDP based?
00124         if (chain[0] == PT_TCP) {
00125                 //3. This is a tcp acceptor. Instantiate it and start accepting connections
00126                 TCPAcceptor *pAcceptor = new TCPAcceptor(node[CONF_IP],
00127                                 node[CONF_PORT], node, chain);
00128                 if (!pAcceptor->Bind()) {
00129                         FATAL("Unable to fire up acceptor from this config node: %s",
00130                                         STR(node.ToString()));
00131                         return false;
00132                 }
00133                 ADD_VECTOR_END(acceptors, pAcceptor);
00134                 return true;
00135         } else if (chain[0] == PT_UDP) {
00136                 //4. Ok, this is an UDP acceptor. Because of that, we can instantiate
00137                 //the full stack. Get the stack first
00138                 BaseProtocol *pProtocol = ProtocolFactoryManager::CreateProtocolChain(
00139                                 chain, node);
00140                 if (pProtocol == NULL) {
00141                         FATAL("Unable to instantiate protocol stack %s", STR(node[CONF_PROTOCOL]));
00142                         return false;
00143                 }
00144 
00145                 //5. Create the carrier and bind it
00146                 UDPCarrier *pUDPCarrier = UDPCarrier::Create(node[CONF_IP], node[CONF_PORT],
00147                                 pProtocol);
00148                 if (pUDPCarrier == NULL) {
00149                         FATAL("Unable to instantiate UDP carrier on %s:%hu",
00150                                         STR(node[CONF_IP]), (uint16_t) node[CONF_PORT]);
00151                         pProtocol->EnqueueForDelete();
00152                         return false;
00153                 }
00154                 pUDPCarrier->SetParameters(node);
00155                 ADD_VECTOR_END(acceptors, pUDPCarrier);
00156 
00157                 //6. We are done
00158                 return true;
00159         } else {
00160                 FATAL("Invalid carrier type");
00161                 return false;
00162         }
