00064         //0. Save the last known address

00066                 _lastAddress = *pPeerAddress;

00078         len = (len + 1)*4;

00080                 WARN("Invalid RTCP packet length: len %hu; bufferLength: %u", len, bufferLength);

00084 
00085         switch (PT) {
00086                 case 200: //SR
00087                 {
00088                         if (len < 28) {
00089                                 WARN("Invalid RTCP packet length: %hu", len);
00090                                 buffer.IgnoreAll();

00092                         }
00093                         uint32_t ntpSec = ENTOHLP(pBuffer + 8) - 2208988800UL;
00094                         uint32_t ntpFrac = ENTOHLP(pBuffer + 12);
00095                         uint64_t ntpMicroseconds = (uint32_t) (((double) ntpFrac / (double) (0x100000000LL))*1000000.0);
00096                         ntpMicroseconds += ((uint64_t) ntpSec)*1000000;
00097                         uint32_t rtpTimestamp = ENTOHLP(pBuffer + 16);
00098                         _pConnectivity->ReportSR(ntpMicroseconds, rtpTimestamp, _isAudio);
00099                         break;
00100                 }
00101                 default:
00102                 {
00103                         WARN("Unknown packet type: %hhu", PT);
00104                         buffer.IgnoreAll();
00105                         return true;
00106                 }
00107         }
00108 
00109         if (pBuffer[1] != 200)
00110                 return true;
00111 
00112         _lsr = ENTOHLP(pBuffer + 10);
00113         buffer.IgnoreAll();

00115         //2. Send the RR
