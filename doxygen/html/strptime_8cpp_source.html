00104 match_string (const char **buf, const char **strs)
00105 {
00106     int i = 0;
00107 
00108     for (i = 0; strs[i] != NULL; ++i) {
00109         int len = strlen (strs[i]);
00110 
00111         if (strncasecmp (*buf, strs[i], len) == 0) {
00112             *buf += len;
