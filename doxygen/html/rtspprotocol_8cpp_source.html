00506 bool RTSPProtocol::ParseNormalHeaders(IOBuffer &buffer) {
00507         _inboundHeaders.Reset();
00508         _inboundContent = "";
00509         //1. We have to have at least 4 bytes (double \r\n)
00510         if (GETAVAILABLEBYTESCOUNT(buffer) < 4) {
00511                 return true;
00512         }
00513 
00514         //2. Detect the headers boundaries
00515         uint32_t headersSize = 0;
00516         bool markerFound = false;
00517         uint8_t *pBuffer = GETIBPOINTER(buffer);
00518         for (uint32_t i = 0; i <= GETAVAILABLEBYTESCOUNT(buffer) - 4; i++) {
00519                 if ((pBuffer[i] == 0x0d)
00520                                 && (pBuffer[i + 1] == 0x0a)
00521                                 && (pBuffer[i + 2] == 0x0d)
00522                                 && (pBuffer[i + 3] == 0x0a)) {
00523                         markerFound = true;
00524                         headersSize = i;
00525                         break;
00526                 }
00527                 if (i >= RTSP_MAX_HEADERS_SIZE) {
00528                         FATAL("Headers section too long");
00529                         return false;
00530                 }
00531         }
00532 
00533         //3. Are the boundaries correct?
00534         //Do we have enough data to parse the headers?
00535         if (headersSize == 0) {
00536                 if (markerFound)
00537                         return false;
00538                 else
00539                         return true;
00540         }
00541 
00542         //4. Get the raw headers and plit it into lines
00543         string rawHeaders = string((char *) GETIBPOINTER(buffer), headersSize);
00544         vector<string> lines;
00545         split(rawHeaders, "\r\n", lines);
00546         if (lines.size() == 0) {
00547                 FATAL("Incorrect RTSP request");
00548                 return false;
00549         }
00550

00552         //for a previous request made by us, or the request that we just received
00553         if (!ParseFirstLine(lines[0])) {
00554                 FATAL("Unable to parse the first line");
00555                 return false;
00556         }
00557
