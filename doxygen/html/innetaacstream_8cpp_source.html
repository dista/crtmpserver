00118 bool InNetAACStream::InitStreamCapabilities(uint8_t *pData, uint32_t dataLength) {
00119         if (_capabilities.audioCodecId == CODEC_AUDIO_UNKNOWN) {
00120                 uint8_t mpegts2rtmpProfile[] = {
00121                         1, 2, 3
00122                 };
00123 
00124                 BitArray codecSetup;
00125                 //profile_index from MPEG-TS different from profile_index from RTMP
00126                 //so we need a map
00127                 uint8_t profile = pData[2] >> 6;
00128                 codecSetup.PutBits<uint8_t > (mpegts2rtmpProfile[profile], 5);
00129 
00130                 //frequence mapping is the same
00131                 //iso13818-7.pdf page 46/206
00132                 uint8_t sampling_frequency_index = (pData[2] >> 2)&0x0f;
00133                 codecSetup.PutBits<uint8_t > (sampling_frequency_index, 4);
00134 
00135                 //this could be an issue... AAC from RTMP only supports stereo
00136                 //and we have mono in this file... Need to check out this one...
00137                 codecSetup.PutBits<uint8_t > (2, 4);
00138 
00139                 _capabilities.InitAudioAAC(GETIBPOINTER(codecSetup),
00140                                 GETAVAILABLEBYTESCOUNT(codecSetup));
00141         }
00142         return true;
