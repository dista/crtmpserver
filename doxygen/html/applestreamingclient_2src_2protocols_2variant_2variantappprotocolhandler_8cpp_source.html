00060 bool VariantAppProtocolHandler::ProcessMessage(BaseVariantProtocol *pProtocol,
00061                 Variant &lastSent, Variant &lastReceived) {
00062         string type = ASC_REQ_TYPE(lastReceived);
00063         if (type == ASC_REQ_TYPE_CONTEXT_CREATE) {
00064                 ProcessContextCreate(pProtocol, lastReceived);
00065         } else if (type == ASC_REQ_TYPE_CONTEXT_LIST) {
00066                 ProcessContextList(pProtocol, lastReceived);
00067         } else if (type == ASC_REQ_TYPE_CONTEXT_CLOSE) {
00068                 ProcessContextClose(pProtocol, lastReceived);
00069         } else if (type == ASC_REQ_TYPE_CONTEXT_CLOSE_ALL) {
00070                 ProcessContextCloseAll(pProtocol, lastReceived);
00071         } else if (type == ASC_REQ_TYPE_COMMAND_PLAY) {
00072                 ProcessCommandPlay(pProtocol, lastReceived);
00073         } else if (type == ASC_REQ_TYPE_COMMAND_SET_BITRATES) {
00074                 ProcessCommandSetBitrates(pProtocol, lastReceived);
00075         } else if (type == ASC_REQ_TYPE_COMMAND_PAUSE) {
00076                 ProcessCommandPause(pProtocol, lastReceived);
00077         } else if (type == ASC_REQ_TYPE_COMMAND_RESUME) {
00078                 ProcessCommandResume(pProtocol, lastReceived);
00079         } else if (type == ASC_REQ_TYPE_INFO_LIST_STREAMS) {
00080                 ProcessInfoListStreams(pProtocol, lastReceived);
00081         } else if (type == ASC_REQ_TYPE_INFO_LIST_ALL_STREAMS) {
00082                 ProcessInfoListAllStreams(pProtocol, lastReceived);
00083         } else if (type == ASC_REQ_TYPE_INFO_BANDWIDTH) {
00084                 ProcessInfoBandwidth(pProtocol, lastReceived);
00085         } else if (type == ASC_REQ_TYPE_INFO_PLAYBACK) {
00086                 ProcessInfoPlayback(pProtocol, lastReceived);
00087         } else {
00088                 WARN("Processing type `%s` not yet implemented", STR(type));
00089                 ASC_RES_BUILD_UNKNOWN_REQUEST_TYPE(lastReceived);
00090         }
00091         return pProtocol->Send(lastReceived);
00092 }
