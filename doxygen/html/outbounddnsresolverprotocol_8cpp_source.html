00083         //1. Build the request

00085         request[DNS_RESOLVER_REQUEST_ID] = _requestIdGenerator++;
00086         request[DNS_RESOLVER_HOST] = host;
00087         request[DNS_RESOLVER_CUSTOM_DATA] = customData;
00088 
00089         //2. Get our hands on the protocol
00090         BaseProtocol *pProtocol = NULL;
00091         if (_dnsProtocolResolverId != 0)
00092                 pProtocol = ProtocolManager::GetProtocol(_dnsProtocolResolverId);
00093 
00094         //3. See if we have a valid protocol
00095         if (pProtocol != NULL) {
00096                 //3.1. We have a protocol. See if it is the correct type
00097                 if (pProtocol->GetType() != PT_OUTBOUND_DNS) {
00098                         //3.1.1 the protocol is not the correct type. Reset the protocol
00099                         //info and queue the request
00100                         _dnsProtocolResolverId = 0;
00101                         pProtocol = NULL;
00102                         _requests[request[DNS_RESOLVER_REQUEST_ID]] = request;
00103                 } else {
00104                         //3.1.2 All is ok. Send the request
00105                         if (!((OutboundDNSResolverProtocol *) pProtocol)->SendRequest(request)) {
00106                                 FATAL("Unable to send requests");
00107                                 return false;

00109                 }
00110         } else {
00111                 //3.2. The dns resolver is not yet available. Queue the request
00112                 _requests[request[DNS_RESOLVER_REQUEST_ID]] = request;
