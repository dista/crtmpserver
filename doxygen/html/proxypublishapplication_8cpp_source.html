00192         //FINEST("parameters:\n%s", STR(parameters.ToString()));

00196                         && ((bool)parameters["customParameters"]["localStreamConfig"]["keepAlive"])
00197                         && (parameters.HasKeyChain(_V_NUMERIC, true, 3, "customParameters", "localStreamConfig", "localUniqueStreamId"))) {
00198                 //3. This is a push.  First, fix the uri
00199                 string uri = parameters["customParameters"]["localStreamConfig"]["targetUri"]["fullUriWithAuth"];
00200                 parameters["customParameters"]["localStreamConfig"]["targetUri"] = uri;
00201                 EnqueuePush(parameters["customParameters"]["localStreamConfig"]);
00202         } else if ((parameters.HasKeyChain(V_BOOL, true, 3, "customParameters", "externalStreamConfig", "keepAlive"))
00203                         && ((bool)parameters["customParameters"]["externalStreamConfig"]["keepAlive"])) {
00204                 //4. This is a pull with keep alive. Just o it again. First, fix the uri
00205                 //which is currently "resolved"

00207                 parameters["customParameters"]["externalStreamConfig"]["uri"] = uri;
00208                 EnqueuePull(parameters["customParameters"]["externalStreamConfig"]);
00209         }
00210 
00211         //5. Finally, call the base class
00212         BaseClientApplication::UnRegisterProtocol(pProtocol);
