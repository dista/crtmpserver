00077 bool SOManager::ProcessSharedObject(BaseRTMPProtocol *pFrom, Variant &request) {
00078         //1. Get the name and the persistance property
00079         string name = M_SO_NAME(request);
00080 
00081         //2. Get or create the SO
00082         SO *pSO = GetSO(name, M_SO_PERSISTANCE(request));
00083 
00084         //3. Hit the SO with the operations requested
00085         for (uint32_t i = 0; i < M_SO_PRIMITIVES(request).MapSize(); i++) {
00086                 if (!ProcessSharedObjectPrimitive(pFrom, pSO, name, request, i)) {
00087                         FATAL("Unable to process primitive %u from\n%s", i,
00088                                         STR(request.ToString()));
00089                         return false;
00090                 }
00091         }
00092 
00093         //4. Get the SO again, but ONLY is it is stil alive
00094         if (MAP_HAS1(_sos, name))
00095                 pSO = _sos[name];
00096         else
00097                 pSO = NULL;
00098 
00099         //5. Track it if is alive
00100         if (pSO != NULL)
00101                 pSO->Track();
00102 
00103         //6. Done
00104         return true;
