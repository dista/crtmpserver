00070 bool VMAppApplication::Initialize() {
00071         //1. Normalize the path to the script
00072         if ((!_configuration.HasKey("script"))
00073                         || (_configuration["script"] != V_STRING)
00074                         || (_configuration["script"] == "")) {
00075                 FATAL("script value provided in the config file is incorrect");
00076                 return false;
00077         }
00078         string script = _configuration["script"];
00079         if (script[0] != '/') {
00080                 script = normalizePath(_configuration[CONF_APPLICATION_DIRECTORY], script);
00081         }
00082         if (script == "") {
00083                 FATAL("file `%s` not found", STR(_configuration["script"]));
00084                 return false;
00085         }
00086         _configuration["script"] = script;
00087 
00088         //2. Get the virtual machine type
00089         if ((!_configuration.HasKey("vmType"))
00090                         || (_configuration["vmType"] != V_STRING)
00091                         || (_configuration["vmType"] == "")) {
00092                 FATAL("vmType value provided in the config file is incorrect");
00093                 return false;
00094         }
00095         _configuration["vmType"] = lowerCase(_configuration["vmType"]);
00096 
00097         //3. Create the virtual machine
00098         _pAppVM = BaseAppVirtualMachine::GetVM(_configuration["vmType"],
00099                         _configuration["script"], this);
00100         if (_pAppVM == NULL) {
00101                 FATAL("Unable to get the %s virtual machine with script %s",
00102                                 STR(_configuration["vmType"]),
00103                                 STR(_configuration["script"]));
00104                 return false;
00105         }
00106 
00107         //4. Plug in the handlers
00108         _pRTMPHandler = new RTMPAppProtocolHandler(_configuration, _pAppVM);
00109         RegisterAppProtocolHandler(PT_INBOUND_RTMP, _pRTMPHandler);
00110         RegisterAppProtocolHandler(PT_INBOUND_RTMPS_DISC, _pRTMPHandler);
00111         RegisterAppProtocolHandler(PT_OUTBOUND_RTMP, _pRTMPHandler);
00112         _pRTPHandler = new RTPAppProtocolHandler(_configuration);
00113         RegisterAppProtocolHandler(PT_INBOUND_RTP, _pRTPHandler);
00114         RegisterAppProtocolHandler(PT_RTCP, _pRTPHandler);
00115         _pRTSPHandler = new RTSPAppProtocolHandler(_configuration);
00116         RegisterAppProtocolHandler(PT_RTSP, _pRTSPHandler);
00117         _pTSHandler = new TSAppProtocolHandler(_configuration, _pAppVM);
00118         RegisterAppProtocolHandler(PT_INBOUND_TS, _pTSHandler);
00119 
00120         if (_pAppVM->HasFunction(handlers_application_initCompleted)) {
00121                 return _pAppVM->Call(handlers_application_initCompleted);
00122         }
00123 
00124         //5. Done
00125         return true;
00126 }
