00069                 sockaddr_in *pPeerAddress) {
00070         //1. Get the raw buffer and its length
00071         uint8_t *pBuffer = GETIBPOINTER(buffer);
00072         uint32_t length = GETAVAILABLEBYTESCOUNT(buffer);
00073 
00074         //2. Do we have enough data?
00075         if (length < 12) {
00076                 buffer.IgnoreAll();
00077                 return true;
00078         }
00079 
00080         //3. Get the RTP header parts that we are interested in
00081         _rtpHeader._flags = ENTOHLP(pBuffer);
00082         _rtpHeader._timestamp = ENTOHLP(pBuffer + 4);
00083         _rtpHeader._ssrc = ENTOHLP(pBuffer + 8);
00084 
00085         //4. Advance the sequence roll-over counter if necessary
00086         if (GET_RTP_SEQ(_rtpHeader) < _lastSeq) {
00087                 if ((_lastSeq - GET_RTP_SEQ(_rtpHeader)) > (0xffff >> 2)) {
00088                         _seqRollOver++;
00089                         _lastSeq = GET_RTP_SEQ(_rtpHeader);
00090                 } else {
00091                         buffer.IgnoreAll();
00092                         return true;
00093                 }
00094         } else {
00095                 _lastSeq = GET_RTP_SEQ(_rtpHeader);
00096         }
00097 
00098         //5. Do we have enough data?
00099         if (length < ((uint32_t) 12 + GET_RTP_CC(_rtpHeader)*4 + 1)) {
00100                 buffer.IgnoreAll();
00101                 return true;
00102         }
00103 
00104         //6. Skip the RTP header
00105         pBuffer += 12 + GET_RTP_CC(_rtpHeader)*4;
00106         length -= 12 + GET_RTP_CC(_rtpHeader)*4;
00107 
00108         //7. Detect rollover and adjust the timestamp
00109 #ifdef RTP_DETECT_ROLLOVER
00110         if (_rtpHeader._timestamp < _lastTimestamp) {
00111                 if ((((_rtpHeader._timestamp & 0x80000000) >> 31) == 0)
00112                                 && (((_lastTimestamp & 0x80000000) >> 31) == 1)) {
00113                         _timestampRollover++;
00114                         _lastTimestamp = _rtpHeader._timestamp;
00115                 }
00116         } else {
00117                 _lastTimestamp = _rtpHeader._timestamp;
00118         }
00119         _rtpHeader._timestamp = (_timestampRollover << 32) | _rtpHeader._timestamp;
00120 #endif
00121 
00122         //5. Feed the data to the stream
00123         if (_pInStream != NULL) {
00124                 if (_isAudio) {
00125                         if (!_pInStream->FeedAudioData(pBuffer, length, _rtpHeader)) {
00126                                 FATAL("Unable to stream data");
00127                                 if (_pConnectivity != NULL) {
00128                                         _pConnectivity->EnqueueForDelete();
00129                                         _pConnectivity = NULL;
00130                                 }
00131                                 return false;
00132                         }
00133                 } else {
00134                         if (!_pInStream->FeedVideoData(pBuffer, length, _rtpHeader)) {
00135                                 FATAL("Unable to stream data");
00136                                 if (_pConnectivity != NULL) {
00137                                         _pConnectivity->EnqueueForDelete();
00138                                         _pConnectivity = NULL;
00139                                 }
00140                                 return false;
00141                         }
00142                 }
00143         }
00144 
00145         //6. Ignore the data
00146         buffer.IgnoreAll();
00147 
00148         //7. Increment the packets count
00149         _packetsCount++;
00150 
00151         //8. Send the RR if necesary
00152         if ((_packetsCount % 300) == 0) {
00153

00155                         if (!_pConnectivity->SendRR(_isAudio)) {
00156                                 FATAL("Unable to send RR");
