00311 bool ClientContext::ConsumeAVBuffer() {

00313         if (_lastWallClock == 0) {
00314 #ifdef HAS_MS_TIMER
00315                 GETCLOCKS(_lastWallClock);
00316 #else
00317                 _lastWallClock = time(NULL)*1000.0;
00318 #endif /* HAS_MS_TIMER */
00319                 return true;
00320         }
00321 
00322         //2. Get the TS protocol
00323         InboundTSProtocol *pTS = (InboundTSProtocol *) ProtocolManager::GetProtocol(_tsId);
00324         if (pTS == NULL) {
00325                 WARN("No TS protocol");
00326                 return true;
00327         }
00328 
00329         //3. First, feed up until the stream name is registered
00330         if (_streamId == 0) {
00331                 while ((_streamId == 0) && (GETAVAILABLEBYTESCOUNT(_avData) > 8192)) {
00332                         if (!pTS->SignalInputData(_avData)) {
00333                                 FATAL("Unable to feed TS protocol");
00334                                 return false;
00335                         }
00336                 }
00337         }
00338 
00339 
00340         //4. Get the inbound TS stream
00341         if (_pStreamsManager == NULL) {
00342                 WARN("No stream manager yet");
00343                 return true;
00344         }
00345         InNetTSStream *pStream = (InNetTSStream *) _pStreamsManager->FindByUniqueId(
00346                         _streamId);
00347         if (pStream == NULL) {
00348                 FATAL("Unable to get the inbound stream");
00349                 return false;
00350         }
00351 
00352         //5. Continue feeding until we have stream capabilities
00353         if ((pStream->GetCapabilities()->videoCodecId != CODEC_VIDEO_AVC)
00354                         || (pStream->GetCapabilities()->audioCodecId != CODEC_AUDIO_AAC)) {
00355                 while (GETAVAILABLEBYTESCOUNT(_avData) > 8192) {
00356                         if ((pStream->GetCapabilities()->videoCodecId == CODEC_VIDEO_AVC)
00357                                         && (pStream->GetCapabilities()->audioCodecId == CODEC_AUDIO_AAC)) {
00358                                 _pEventSink->SignalStreamRegistered(_streamName);
00359                                 break;
00360                         }
00361                         if (!pTS->SignalInputData(_avData)) {
00362                                 FATAL("Unable to feed TS protocol");
00363                                 return false;
00364                         }
00365                 }
00366                 if ((pStream->GetCapabilities()->videoCodecId != CODEC_VIDEO_AVC)
00367                                 || (pStream->GetCapabilities()->audioCodecId != CODEC_AUDIO_AAC)) {
00368                         FINEST("SPS/PPS not yet available");
00369                         return true;
00370                 }
00371         }
00372 
00373         //5. Does it have any registered consumers?
00374         if (pStream->GetOutStreams().size() <= 0) {
00375                 WARN("No registered consumers or stream capabilities not yet known. Take a break...");
00376                 return true;
00377         }
00378 
00379         //5. How much seconds do we have to feed?
00380 #ifdef HAS_MS_TIMER
00381         double currentTime;
00382         GETCLOCKS(currentTime);
00383         double wallClockDelta = ((currentTime - _lastWallClock) / (double) CLOCKS_PER_SECOND)*1000.00;
00384 #else
00385         double wallClockDelta = time(NULL)*1000.00 - _lastWallClock;
00386 #endif /* HAS_MS_TIMER */
00387 
00388         if (_firstFeedTime == 0) {
00389                 _firstFeedTime = pStream->GetFeedTime();
00390         }
00391 
00392         //6. Feed
00393         while ((wallClockDelta + 1000 > (pStream->GetFeedTime() - _firstFeedTime)) &&
00394                         (GETAVAILABLEBYTESCOUNT(_avData) > 8192)) {
00395                 if (!pTS->SignalInputData(_avData)) {
00396                         FATAL("Unable to feed TS protocol");
00397                         return false;
00398                 }
00399         }
