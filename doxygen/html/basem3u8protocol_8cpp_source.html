00077                         GETAVAILABLEBYTESCOUNT(buffer))) {

00084         if (!SignalPlaylistAvailable()) {
00085                 FATAL("Unable to signal playlist available");
00086                 return false;
00087         }
00088 
00089         //6. Job done. Disconnect
00090         EnqueueForDelete();
