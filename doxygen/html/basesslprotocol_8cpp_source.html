00133         BIO *pInBio = SSL_get_rbio(_pSSL);
00134 
00135         //2. dump all the data from the network inside the ssl input
00136         BIO_write(pInBio, GETIBPOINTER(buffer),
00137                         GETAVAILABLEBYTESCOUNT(buffer));

00141         if (!_sslHandshakeCompleted) {
00142                 if (!DoHandshake()) {
00143                         FATAL("Unable to do the SSL handshake");
00144                         return false;
00145                 }
00146                 if (!_sslHandshakeCompleted) {
00147                         return true;

00152         int32_t read = 0;
00153         while ((read = SSL_read(_pSSL, _pReadBuffer, MAX_SSL_READ_BUFFER)) > 0) {
00154                 _inputBuffer.ReadFromBuffer(_pReadBuffer, (uint32_t) read);
00155         }

00159                                 error != SSL_ERROR_WANT_WRITE) {
00160                         FATAL("Unable to read data: %d", error);
00161                         return false;
00162                 }
00163         }
00164 
00165         //6. If we have pending data inside the decrypted buffer, bubble it up on the protocol stack
00166         if (GETAVAILABLEBYTESCOUNT(_inputBuffer) > 0) {
00167                 if (_pNearProtocol != NULL) {
00168                         if (!_pNearProtocol->SignalInputData(_inputBuffer)) {

00176         //data that needs to be sent.
