00076         //1. Get the HTTP protocol
00077         OutboundHTTPProtocol *pHTTP = (OutboundHTTPProtocol *) GetFarProtocol();
00078 
00079         //2. Is the transfer completed?

00086         string data = string((const char *) GETIBPOINTER(buffer), GETAVAILABLEBYTESCOUNT(buffer));
00087         FINEST("data:\n%s", STR(data));
00088 
00089         //4. consume the buffer
00090         buffer.IgnoreAll();

00092         //5. We are done
00093         return true;
