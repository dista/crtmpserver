00039         //1. Test the wanted name

00065         if (pApp->GetId() == GetApplication()->GetId()) {
00066                 FATAL("appselector can be a final destination");
00067                 return false;

00069 
00070         //5. Get the RTMP handler from that app
00071         BaseRTMPAppProtocolHandler *pHandler =

00073         if (pHandler == NULL) {
00074                 FATAL("Unable to get a valid handler");
00075                 return false;

00077

00081                 FINEST("Selected application: %s", STR(appName));

00083                 FINEST("Selected application: %s (%s)", STR(realAppName), STR(appName));
00084 
00085         //7. Set the new application inside the connection

00087
