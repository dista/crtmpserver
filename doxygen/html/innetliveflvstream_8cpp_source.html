00152 bool InNetLiveFLVStream::SendStreamMessage(Variant &completeMessage, bool persistent) {
00153         //2. Loop on the subscribed streams and send the message
00154         LinkedListNode<BaseOutStream *> *pTemp = _pOutStreams;
00155         while ((pTemp != NULL) && (!IsEnqueueForDelete())) {
00156                 if (pTemp->info->IsEnqueueForDelete()) {
00157                         continue;
00158                 }
00159                 if (TAG_KIND_OF(pTemp->info->GetType(), ST_OUT_NET_RTMP)) {
00160                         if (!((BaseOutNetRTMPStream *) pTemp->info)->SendStreamMessage(completeMessage)) {
00161                                 FATAL("Unable to send notify on stream. The connection will go down");
00162                                 pTemp->info->EnqueueForDelete();
00163                         }
00164                 }
00165                 pTemp = pTemp->pPrev;
00166         }
00167 
00168         //3. Test to see if we are still alive. One of the target streams might
00169         //be on the same RTMP connection as this stream is and our connection
00170         //here might be enque for delete
00171         if (IsEnqueueForDelete())
00172                 return false;
00173 
00174         //4. Save the message for future use if necessary
00175         if (persistent)
00176                 _lastStreamMessage = completeMessage;
00177 
00178         //5. Done
00179         return true;
