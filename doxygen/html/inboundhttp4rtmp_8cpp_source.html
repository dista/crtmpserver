00065 bool InboundHTTP4RTMP::SignalInputData(IOBuffer &buffer) {
00066         //1. Get the HTTP far protool and test to see if it has ContentLength
00067         InboundHTTPProtocol *pHTTP = (InboundHTTPProtocol *) _pFarProtocol;
00068         if (pHTTP == NULL || pHTTP->GetContentLength() == 0) {
00069                 FATAL("Invalid HTTP request");
00070                 return false;
00071         }
00072 
00073         //2. Test it and see if all the data was transfered
00074         if (!pHTTP->TransferCompleted()) {
00075                 return true;
00076         }
00077 
00078         //3. Get the HTTP request
00079         Variant request = pHTTP->GetHeaders();
00080 
00081         //4. Is this a keep-alive?
00082         pHTTP->SetDisconnectAfterTransfer(
00083                         request[HTTP_HEADERS][HTTP_HEADERS_CONNECTION]
00084                         != HTTP_HEADERS_CONNECTION_KEEP_ALIVE);
00085         DeleteNearProtocol(false);
00086 
00087         //4. Get the URL
00088         string url = request[HTTP_FIRST_LINE][HTTP_URL];
00089 
00090         //5. split it in meaningful parts
00091         vector<string> parts;
00092         split(url, "/", parts);
00093         if (parts.size() < 2) {
00094                 FATAL("Invalid request:\n%s", STR(request.ToString()));
00095                 return false;
00096         }
00097 
00098         //7. Do the dammage
00099         bool result;
00100         if (parts[1] == "fcs") {
00101                 result = ProcessFcs(parts);
00102                 buffer.Ignore(pHTTP->GetContentLength());
00103         } else if (parts[1] == "open") {
00104                 result = ProcessOpen(parts);
00105                 buffer.Ignore(pHTTP->GetContentLength());
00106         } else if (parts[1] == "idle") {
00107                 result = ProcessIdle(parts);
00108                 buffer.Ignore(pHTTP->GetContentLength());
00109         } else if (parts[1] == "send") {
00110                 if (GETAVAILABLEBYTESCOUNT(buffer) < 1)
00111                         return false;
00112                 _inputBuffer.ReadFromBuffer(GETIBPOINTER(buffer), pHTTP->GetContentLength());
00113                 buffer.Ignore(pHTTP->GetContentLength());
00114                 result = ProcessSend(parts);
00115         } else {
00116                 FATAL("Invalid command: %s", STR(parts[1]));
00117                 result = false;
00118         }
00119 
00120         //8. Cleanup
00121         if (!result) {
00122                 DeleteNearProtocol(true);
00123                 EnqueueForDelete();
00124         }
00125 
00126         //9. Done
00127         return result;
00128 }
