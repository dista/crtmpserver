00084                 double absoluteTimestamp) {

00088         //6.2  Audio Data Transport Stream, ADTS
00089         //iso13818-7 page 26/206

00099                         codecSetup.ReadFromBuffer(pCapabilities->aac._pAAC,
00100                                         pCapabilities->aac._aacLength);

00107                                         absoluteTimestamp, //absoluteTimestamp

00110                                 FATAL("Unable to send audio codec setup");
00111                                 return false;
00112                         }
00113                 }

00116         }
00117

00119                 pData[0] = 0xaf;
00120                 pData[1] = 0x01;
00121

00123                                 pData, //pData
00124                                 dataLength, //dataLength
00125                                 0, //processedLength
00126                                 dataLength, //totalLength
00127                                 absoluteTimestamp, //absoluteTimestamp
00128                                 true //isAudio
